  LOCAL OCR NORMALIZATION FUNCTION (private, at file scope, below imports):
    private fun normalizeOcrText(raw: String): String
    Normalization pipeline (OCR-03) applied in this order:
    1. Rupee symbol variants: replace all of "Rs.", "Rs ", "INR ", "rs.", "₹ " (with trailing space), "RS." with "₹"
    2. Digit-letter confusion in numeric contexts:
       - In a sequence of digits, replace letter 'O' or 'o' with '0' (regex: "(?<=[0-9])[Oo](?=[0-9])" or position-based)
       - Replace letter 'l' or 'I' with '1' when surrounded by digits (regex: "(?<=[0-9])[lI](?=[0-9])")
    3. Thousands separators: normalize "1,00,000" style (Indian) — leave commas in place (they are valid), normalize any stray dots used as thousands separators in non-decimal contexts
    4. Boilerplate stripping: remove lines matching common receipt header/footer patterns:
       - Lines equal to (case-insensitive): "thank you", "thank you for shopping", "customer copy", "merchant copy", "original", "duplicate"
       - Lines containing only dashes, asterisks, or equals signs (e.g., "---", "***", "===")
    Return the cleaned string with trimmed leading/trailing whitespace. Preserve all other content.

  OCR TRIGGER FUNCTION (inline coroutineScope.launch in triggerOcr, or as a suspend lambda):
     fun triggerOcr(uri: Uri) {
       coroutineScope.launch {
         isOcrProcessing = true
         ocrText = null
         try {
           val recognizedText = withContext(Dispatchers.IO) {
             val inputImage = InputImage.fromFilePath(context, uri)
             val recognizer = TextRecognition.getClient(TextRecognizerOptions.DEFAULT_OPTIONS)
             // Use Tasks.await() or suspendCoroutine to bridge the Task<Text> callback:
             val result = com.google.android.gms.tasks.Tasks.await(recognizer.process(inputImage))
             result.text
           }
           val normalized = normalizeOcrText(recognizedText)
           ocrText = normalized
           ocrFailureCount = 0
         } catch (e: Exception) {
           ocrFailureCount++
           // ocrText remains null — UI shows retry or manual entry based on ocrFailureCount
         } finally {
           isOcrProcessing = false
         }
       }
     }
