  RECORDING FUNCTIONS (defined as local lambdas or inner functions within the composable scope):
    fun startRecording() {
      val recognizer = SpeechRecognizer.createSpeechRecognizer(context)
      speechRecognizer = recognizer
      recognizer.setRecognitionListener(object : RecognitionListener {
        override fun onReadyForSpeech(params: Bundle?) { isRecording = true }
        override fun onRmsChanged(rmsdB: Float) {
          // Normalize rmsdB (typically -2 to 10 dB range) to 0..1
          val normalized = ((rmsdB + 2f) / 12f).coerceIn(0f, 1f)
          // Update rolling buffer: drop oldest, append newest
          if (rmsBuffer.size >= 7) rmsBuffer.removeAt(0)
          rmsBuffer.add(normalized)
        }
        override fun onResults(results: Bundle?) {
          val matches = results?.getStringArrayList(SpeechRecognizer.RESULTS_RECOGNITION)
          transcription = matches?.firstOrNull() ?: ""
          isRecording = false
          hasRecorded = true
          // Reset waveform
          rmsBuffer.replaceAll { 0f }
        }
        override fun onError(error: Int) {
          isRecording = false
          hasRecorded = true
          if (transcription.isEmpty()) {
            transcription = ""  // leave empty, hint text will show
          }
          rmsBuffer.replaceAll { 0f }
        }
        override fun onBeginningOfSpeech() {}
        override fun onBufferReceived(buffer: ByteArray?) {}
        override fun onEndOfSpeech() { isRecording = false }
        override fun onPartialResults(partialResults: Bundle?) {}
        override fun onEvent(eventType: Int, params: Bundle?) {}
      })
      val intent = Intent(RecognizerIntent.ACTION_RECOGNIZE_SPEECH).apply {
        putExtra(RecognizerIntent.EXTRA_LANGUAGE_MODEL, RecognizerIntent.LANGUAGE_MODEL_FREE_FORM)
        putExtra(RecognizerIntent.EXTRA_PREFER_OFFLINE, true)
        putExtra(RecognizerIntent.EXTRA_MAX_RESULTS, 1)
      }
      recognizer.startListening(intent)
    }

    fun stopRecording() {
      speechRecognizer?.stopListening()
      isRecording = false
    }

    fun startNewSession() {
      speechRecognizer?.destroy()
      speechRecognizer = null
      transcription = ""
      hasRecorded = false
      elapsedSeconds = 0
      rmsBuffer.replaceAll { 0f }
    }
