   ```kotlin
           title = "Theme",

           trailing = {
               ExposedDropdownMenuBox(
                   expanded = themeDropdownExpanded,
                   onExpandedChange = { themeDropdownExpanded = it }
               ) {
                   Box(
                       Modifier
                           .menuAnchor()
                           .padding(8.dp)
                   ) {
                       Text(
                           text = uiState.selectedTheme.displayName,
                           color = MaterialTheme.colorScheme.primary
                       )
                   }
                   ExposedDropdownMenu(
                       expanded = themeDropdownExpanded,
                       onDismissRequest = { themeDropdownExpanded = false }
                   ) {
                       themeOptions.forEach { theme ->
                           DropdownMenuItem(
                               text = { 
                                   Text(
                                       text = theme.displayName,
                                       fontWeight = if (theme == uiState.selectedTheme) FontWeight.Bold else FontWeight.Normal,
                                       color = if (theme == uiState.selectedTheme) MaterialTheme.colorScheme.primary else MaterialTheme.colorScheme.onSurface
                                   )
                               },
                               onClick = {
                                   viewModel.setSelectedTheme(theme)
                                   themeDropdownExpanded = false
                               }
                           )
                       }
                   }
               }
           }
       )
   }
   ```
